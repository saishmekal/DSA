class Solution:

    def aggressiveCows(self, stalls, k):
        stalls.sort()

        def canPlace(stalls, k, dist):
            count = 1
            last_pos = stalls[0]
            for i in range(1, len(stalls)):
                if stalls[i] - last_pos >= dist:
                    count += 1
                    last_pos = stalls[i]
                if count >= k:
                    return True
            return False

        low, high = 1, stalls[-1] - stalls[0]
        ans = 0

        while low <= high:
            mid = (low + high) // 2
            if canPlace(stalls, k, mid):
                ans = mid
                low = mid + 1
            else:
                high = mid - 1

        return ans
