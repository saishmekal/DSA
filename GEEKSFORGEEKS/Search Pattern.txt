def computeLPSArray(pattern):
n = len(pattern)
lps = [0] * n

# length of the previous longest prefix suffix
len = 0
i =1

while i < n:
if pattern[i] == pattern[len]:
len += 1
lps[i] = len
i +1
else:
if len != 0:
# fall back in the pattern
len = lps[len - 1]
else:
lps[i] = 0
i += 1

return 1ps
