class Solution:
    def findMajority(self, arr):
        n = len(arr)
        freq = {}
        res = []

        for ele in arr:
            freq[ele] = freq.get(ele, 0) + 1

        for ele, cnt in freq.items():
            if cnt > n // 3:
                res.append(ele)

        if len(res) == 2 and res[0] > res[1]:
            res[0], res[1] = res[1], res[0]

        return res
